import { createApp } from '/lib/petite-vue/petite-vue.es.min.js'

createApp({

    // data
    countryCode: null,
    countryInfo: null,
    availableCountries: null,
    publicHolidays: null,
    publicHolidaysWorldwide: null,
    search: null,

    // computed
    get filteredCountries() {
        if (this.search === null) {
            return this.availableCountries
        }

        const search = this.search.toLowerCase()
        const filteredItems = this.availableCountries.filter(o => o.countryCode.toLowerCase().includes(search) || o.name.toLowerCase().includes(search))

        return filteredItems
    },

    // methods
    async loadData() {
        await this.getAvailableCountries()
        this.getCurrentCountryViaBrowser()
        await this.getCountryInfo()
        await this.getNextPublicHolidaysWorldwide()
    },
    getCurrentCountryViaBrowser() {
        for (const language of navigator.languages) {
            const locale = new Intl.Locale(language)
            if (locale.region) {
                this.countryCode = locale.region
                break;
            }
        }

        if (!this.countryCode) {
            this.countryCode = 'en'
        }
    },
    getCountryName(countryCode) {
        if (!this.availableCountries) {
            return
        }
        return this.availableCountries && this.availableCountries.find(o => o.countryCode === countryCode).name
    },
    shortLocalDate(date) {
        return dayjs(date).format('DD MMM YYYY')
    },
    longLocalDate(date) {
        return dayjs(date).format('ddd DD MMM YYYY')
    },
    async getAvailableCountries() {
        const response = await axios.get('/Api/v3/AvailableCountries')
        this.availableCountries = response.data
    },
    async getCountryInfo() {
        const response = await axios.get(`/Api/v3/CountryInfo/${this.countryCode}`)
        this.countryInfo = response.data
        this.getNextPublicHolidays()
    },
    async getNextPublicHolidays() {
        const response = await axios.get(`/Api/v3/NextPublicHolidays/${this.countryInfo.countryCode}`)
        this.publicHolidays = response.data
    },
    async getNextPublicHolidaysWorldwide() {
        const response = await axios.get('/Api/v3/NextPublicHolidaysWorldwide')
        this.publicHolidaysWorldwide = response.data
    }
}).mount('#app')
